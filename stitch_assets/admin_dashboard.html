'' : ' No appointments scheduled for today. '; appointments.forEach(appt => { const timeStr = new Date(appt.scheduled_datetime).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); const isCosmetic = appt.type_name?.toLowerCase().includes('botox') || appt.type_name?.toLowerCase().includes('filler'); const hasHold = appt.payment_info?.paymentStrategy === 'auth_hold' && !appt.payment_info?.captured; const row = document.createElement('tr'); row.className = 'hover:bg-slate-50 dark:hover:bg-white/5 transition-colors'; row.innerHTML = ` ${timeStr} ${appt.provider_name || 'Unassigned'}

${appt.patient_name}

ID: ${appt.patient_id.slice(0, 8)}

${isCosmetic && hasHold ? ` ` : ''} `; body.appendChild(row); }); } async function processNoShow(id) { if (!confirm('Charge no-show deposit for this appointment?')) return; try { const result = await smartbookApi.processNoShow(id); if (result.success) { alert('No-show fee captured'); loadSchedule(); } else { alert('Error: ' + result.error); } } catch (err) { console.error('No-show processing failed:', err); } } function showMergeModal() { document.getElementById('mergeModal').classList.remove('hidden'); } function hideMergeModal() { document.getElementById('mergeModal').classList.add('hidden'); } async function executeMerge() { const keepId = document.getElementById('mergeKeepId').value; const fromId = document.getElementById('mergeFromId').value; if (!keepId || !fromId) return alert('Both IDs required'); try { const result = await smartbookApi.mergePatients(keepId, fromId); if (result.success) { alert('Patients merged successfully'); hideMergeModal(); loadSchedule(); } else { alert('Error: ' + result.error); } } catch (err) { console.error('Merge failed:', err); } } loadDashboardStats(); loadSchedule();
